sed -i "s%https://download.astralinux.ru%http://77.88.8.8%g" /etc/apt/sources.list

apt update

3. Второй вариант скрипта для бэкапа LDAP через ldapsearch:

```
#!/bin/bash
set -euo pipefail

DATE=$(date +%F-%H-%M)
WORKDIR=$(mktemp -d /tmp/ipa_backup.XXXXXX)

ARCHIVE="/mnt/backup-user.backup"
DECRYPTED="$WORKDIR/archive.tar"
KEY_FILE="/root/.backup/backup.key"
LDIF_FILE="$WORKDIR/ldap_$DATE.ldif"

LOG="/var/log/ipa-backup.log"
exec >> "$LOG" 2>&1

echo "[*] $(date) Start backup"

# Проверка ключа
[ -f "$KEY_FILE" ] || { echo "[!] Key missing"; exit 1; }

# 1. Экспорт LDAP (онлайн)
echo "[*] Export LDAP via ldapsearch"

ldapsearch -LLL -Y GSSAPI -b "" -s sub "(objectClass=*)" > "$LDIF_FILE"

[ -s "$LDIF_FILE" ] || { echo "[!] LDIF empty"; exit 1; }

# 2. Если архив существует — расшифровываем
if [ -f "$ARCHIVE" ]; then
    echo "[*] Decrypt existing archive"

    openssl enc -d -aes-256-cbc -pbkdf2 -in "$ARCHIVE" -out "$DECRYPTED" -pass file:"$KEY_FILE"

    mkdir -p "$WORKDIR/extracted"
    tar -xf "$DECRYPTED" -C "$WORKDIR/extracted"
else
    echo "[*] No existing archive, creating new"
    mkdir -p "$WORKDIR/extracted"
fi

# 3. Добавляем новый LDIF
cp "$LDIF_FILE" "$WORKDIR/extracted/"

# 4. Пересобираем архив
echo "[*] Repack archive"
tar -cf "$DECRYPTED" -C "$WORKDIR/extracted" .

# 5. Шифруем обратно
echo "[*] Encrypt archive"
openssl enc -aes-256-cbc -salt -pbkdf2 -in "$DECRYPTED" -out "$ARCHIVE" -pass file:"$KEY_FILE"

# 6. Очистка
rm -rf "$WORKDIR"

echo "[+] Done"
```
